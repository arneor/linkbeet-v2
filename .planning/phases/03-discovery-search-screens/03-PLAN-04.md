import 'package:dartz/dartz.dart';
import 'package:equatable/equatable.dart';

import '../../../../core/domain/errors/failures.dart';
import '../../../../core/domain/usecases/usecase.dart';
import '../entities/profile_entity.dart';
import '../repositories/discovery_repository.dart';

class GetNearbyProfiles extends UseCase<List<ProfileEntity>, NearbyParams> {
  final DiscoveryRepository repository;

  GetNearbyProfiles(this.repository);

  @override
  Future<Either<Failure, List<ProfileEntity>>> call(NearbyParams params) {
    return repository.getNearbyProfiles(
      latitude: params.latitude,
      longitude: params.longitude,
      radiusKm: params.radiusKm,
    );
  }
}

class NearbyParams extends Equatable {
  final double latitude;
  final double longitude;
  final double radiusKm;

  const NearbyParams({
    required this.latitude,
    required this.longitude,
    this.radiusKm = 5.0,
  });

  @override
  List<Object?> get props => [latitude, longitude, radiusKm];
}

**near_me_cubit.dart**:
```dart
import 'package:flutter_bloc/flutter_bloc.dart';
import '../../../domain/usecases/get_nearby_profiles.dart';
import 'near_me_state.dart';

class NearMeCubit extends Cubit<NearMeState> {
  final GetNearbyProfiles _getNearbyProfiles;

  NearMeCubit({
    required GetNearbyProfiles getNearbyProfiles,
  })  : _getNearbyProfiles = getNearbyProfiles,
        super(const NearMeState(status: NearMeStatus.locationRequired));

  /// Simulate granting location permission (mock — always succeeds)
  Future<void> grantLocation() async {
    emit(state.copyWith(
      locationGranted: true,
      status: NearMeStatus.loading,
    ));
    await _loadNearby();
  }

  Future<void> changeRadius(double radius) async {
    emit(state.copyWith(
      selectedRadius: radius,
      status: NearMeStatus.loading,
    ));
    await _loadNearby();
  }

  Future<void> _loadNearby() async {
    final result = await _getNearbyProfiles(NearbyParams(
      latitude: NearMeState.mockLatitude,
      longitude: NearMeState.mockLongitude,
      radiusKm: state.selectedRadius,
    ));

    result.fold(
      (failure) => emit(state.copyWith(
        status: NearMeStatus.error,
        errorMessage: failure.message,
      )),
      (profiles) => emit(state.copyWith(
        status: profiles.isEmpty ? NearMeStatus.empty : NearMeStatus.loaded,
        nearbyProfiles: profiles,
      )),
    );
  }
}
````
