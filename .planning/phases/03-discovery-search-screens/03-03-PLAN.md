class DiscoveryRepositoryImpl implements DiscoveryRepository {

  @override
  Future<Either<Failure, List<ProfileEntity>>> getTrendingProfiles() async {
    try {
      final models = await _datasource.getTrendingProfiles();
      return Right(models.map((m) => m.toEntity()).toList());
    } catch (e) {
      return Left(ServerFailure(message: e.toString()));
    }

  @override
  Future<Either<Failure, SearchResultEntity>> searchProfiles({
    required String query,
    String? categoryFilter,
    int page = 1,
    int limit = 20,

    try {
      final models = await _datasource.searchProfiles(query, categoryFilter: categoryFilter);
      final entities = models.map((m) => m.toEntity()).toList();
      return Right(SearchResultEntity(
        profiles: entities,
        totalCount: entities.length,

    try {
      final models = await _datasource.getNearbyProfiles(radiusKm);

    } catch (e) {

import 'package:flutter_bloc/flutter_bloc.dart';
import '../../../../core/domain/usecases/usecase.dart';
import '../../../data/models/filter_model.dart';
import '../../../domain/entities/filter_entity.dart';
import '../../../domain/usecases/get_trending_profiles.dart';
import 'discovery_state.dart';

class DiscoveryCubit extends Cubit<DiscoveryState> {
  final GetTrendingProfiles _getTrendingProfiles;

  DiscoveryCubit({
    required GetTrendingProfiles getTrendingProfiles,
  })  : _getTrendingProfiles = getTrendingProfiles,
        super(const DiscoveryState());

  Future<void> loadDiscovery() async {
    emit(state.copyWith(
      status: DiscoveryStatus.loading,
      greeting: _getGreeting(),
      filters: FilterModel.defaultFilters,
    ));

    final result = await _getTrendingProfiles(NoParams());

    result.fold(
      (failure) => emit(state.copyWith(
        status: DiscoveryStatus.error,
        errorMessage: failure.message,
      )),
      (profiles) => emit(state.copyWith(
        status: DiscoveryStatus.loaded,
        trendingProfiles: profiles,
      )),
    );
  }

  void toggleFilter(String filterId) {
    final updated = state.filters.map((f) {
      if (f.id == filterId) return f.copyWith(isSelected: !f.isSelected);
      return f;
    }).toList();
    emit(state.copyWith(filters: updated));
  }

  String _getGreeting() {
    final hour = DateTime.now().hour;
    if (hour < 12) return 'Good morning';
    if (hour < 17) return 'Good afternoon';
    return 'Good evening';
  }
}

import 'package:flutter_bloc/flutter_bloc.dart';
import '../../../domain/usecases/search_profiles.dart';
import 'search_state.dart';

class SearchCubit extends Cubit<SearchState> {
  final SearchProfiles _searchProfiles;

  SearchCubit({
    required SearchProfiles searchProfiles,
  })  : _searchProfiles = searchProfiles,
        super(const SearchState());

  Future<void> search(String query, {String? categoryFilter}) async {
    if (query.trim().isEmpty) {
      emit(const SearchState());
      return;
    }

    emit(state.copyWith(status: SearchStatus.loading, query: query));

    final result = await _searchProfiles(SearchParams(
      query: query,
      categoryFilter: categoryFilter,
    ));

    result.fold(
      (failure) => emit(state.copyWith(
        status: SearchStatus.error,
        errorMessage: failure.message,
      )),
      (searchResult) => emit(state.copyWith(
        status: searchResult.profiles.isEmpty ? SearchStatus.empty : SearchStatus.loaded,
        results: searchResult.profiles,
        totalCount: searchResult.totalCount,
      )),
    );
  }

  void clearSearch() {
    emit(const SearchState());
  }
}
